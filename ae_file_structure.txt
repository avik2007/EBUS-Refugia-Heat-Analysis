│   ├── 03b_ae_plot_physics.py            # Regenerates physics PNGs from the 03 audit

│   ├── 04b_ae_plot_matern_physics.py     # Regenerates physics PNGs from both 04

│   │
│   ├── 05_ae_rmsre_optimization.py       # RMSRE optimization experiment. Ran 4 variants
│   │                                      #   (C1–C4) against the C0 baseline to find the
│   │                                      #   best config for the Skin Layer. Winner: C2
│   │                                      #   (window=45d, 3D Matern). Results in aelogs/
│   │                                      #   under _3d_m05_w* and _minbins80 folders.
│   │
│   ├── 05_ae_update_tomatern0.5.py       # CANONICAL Skin Layer script (supersedes 03).
│   │                                      #   Adopts C2 config: mode='3D',

│   │                                      #   time_ls_bounds_days=(2.0, 45.0).
│   │                                      #   Output: aelogs/{run_id}_3dmatern_w45/
│   │                                      #   Result: 19/23 pass (83%), median RMSRE 3.86%
│   │
│   ├── 10_ae_backfill_configs.py         # Backfills legacy AEResults/aelogs/ runs into
   │                                      #   configs/<region>/ YAML files. Walks every
   │                                      #   aelogs dir, parses run_id, reconstructs an
   │                                      #   AnalysisConfig, and writes YAML with
   │                                      #   legacy_backfill=true. Unrecoverable fields
   │                                      #   (noise_val, ls_bounds) written as null.
   │                                      #   Output: configs/california/ + configs/californiav2/
   │
   ├── aebus_cli.py                       # Config-driven CLI (aebus). Subcommands:
   │                                      #   validate — schema-check + print run_id
   │                                      #   analyze  — dispatch AnalysisConfig end-to-end
   │                                      #   ingest   — dispatch IngestionConfig end-to-end
   │                                      #   list     — tabulate runs from run_registry.jsonl
   │                                      #   show     — print manifest.json for a run_id
   │                                      #   Exit codes: 0=ok, 2=bad config, 3=collision.
   │
   ├── test_mlops_foundation.py           # 44-test unit suite for the MLOps foundation layer.
   │                                      #   Covers: config_schema (valid + invalid configs,
   │                                      #   cross-field validators, legacy_backfill rules),
   │                                      #   manifest (hash, provenance, collision detect,
   │                                      #   registry append), runner (run_id derivation,
   │                                      #   dispatch dispatch), CLI subcommands.
   │                                      #   Run: pytest ArgoEBUSCloud/test_mlops_foundation.py
   │
   ├── 09_ae_longterm_float_census.py    # Float density census across broad California
│   │                                      #   domain (lat [25,50], lon [-140,-110]) for
│   │                                      #   1999–2025. Fetches via ERDDAP in 5-year
│   │                                      #   chunks, bins on 5°×5° grid (unique floats),
│   │                                      #   saves one Cartopy PNG per year + full CSV.

│   │                                      #   density Cartopy map, and the californiav3
│   │                                      #   domain recommendation (cells present in

│   │                                      #   Output: AEResults/aeplots/float_census_california/

│   ├── ocean_cloud.yml                    # Conda environment spec for cloud workers

│   │
│   └── ebus_core/                         # Shared core library (importable module)
│       ├── __init__.py                    # Makes ebus_core a Python package.
│       │
│       ├── ae_utils.py                    # Configuration registry. Defines lat/lon bounds,
│       │                                  #   S3 bucket paths, and spatial/temporal resolution
│       │                                  #   for each EBUS region (California, Humboldt,
│       │                                  #   Canary, Benguela). Also manages output dirs.
│       │
│       ├── argoebus_thermodynamics.py     # Ocean physics engine. Uses TEOS-10 (GSW library)
│       │                                  #   to convert raw T/S/P into heat energy density
│       │                                  #   (J/m³), then bins and integrates vertically to
│       │                                  #   produce Ocean Heat Content (OHC) in J/m².
│       │
│       ├── argoebus_gp_physics.py         # Gaussian Process Regression engine. Fills spatial
│       │                                  #   gaps in the Argo float coverage using kriging.
│       │                                  #   Auto-tunes correlation lengths via K-Fold or
│       │                                  #   Leave-One-Float-Out cross-validation.
│       │
│       ├── config_schema.py               # Pydantic models for pipeline configs (schema_version 1).
       │                                  #   AnalysisConfig: region, date range, resolution,
       │                                  #   depth range, GPR hyperparams, run_suffix.
       │                                  #   IngestionConfig: same plus CloudBlock, S3Block,
       │                                  #   QCPolicyBlock. Both reject unknown fields and run
       │                                  #   cross-field validators (bin-aliasing, bounds checks).
       │                                  #   legacy_backfill=true skips completeness checks on
       │                                  #   null GPR fields. load_config() auto-selects model.
       │
       ├── manifest.py                    # Manifest capture + collision detection + registry.
       │                                  #   canonical_config_dict() hashes the config for
       │                                  #   run identity. build_manifest() captures git SHA,
       │                                  #   conda env, host, and timing. check_collision()
       │                                  #   detects hash mismatches vs. existing manifests.
       │                                  #   append_registry() writes to run_registry.jsonl.
       │
       ├── runner.py                      # Dispatch wrappers for analysis + ingestion.
       │                                  #   run_analysis(cfg) → calls existing validate_moving_window*
       │                                  #   run_ingestion(cfg) → calls run_ingestion_pipeline()
       │                                  #   Both derive run_id, check collision, write manifest.
       │                                  #   derive_run_id() reproduces the canonical naming scheme
       │                                  #   used by existing AEResults/aelogs/ directories.
       │
       ├── backfill.py                    # Backfill logic for legacy analysis runs.
       │                                  #   _parse_run_id() decomposes a canonical run_id string.
       │                                  #   reconstruct_analysis_config() builds an AnalysisConfig
       │                                  #   YAML from the parsed fields; noise_val and ls_bounds
       │                                  #   are written as null with legacy_backfill=true.
       │                                  #   Called by 10_ae_backfill_configs.py.
       │
       └── argoebus_plotting.py           # Visualization functions. Generates Cartopy maps
│                                          #   of kriging predictions + uncertainty, and
│                                          #   physics history plots over time.
